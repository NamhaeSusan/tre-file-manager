25fn setup_terminal() -> anyhow::Result<Terminal<CrosstermBackend<io::Stdout>>> {
26    enable_raw_mode()?;
27    let mut stdout = io::stdout();
28    execute!(stdout, EnterAlternateScreen)?;
29    let backend = CrosstermBackend::new(stdout);
30    let terminal = Terminal::new(backend)?;
31    Ok(terminal)
32}
33
34fn restore_terminal(terminal: &mut Terminal<CrosstermBackend<io::Stdout>>) -> anyhow::Result<()> {
35    disable_raw_mode()?;
36    execute!(terminal.backend_mut(), LeaveAlternateScreen)?;
37    terminal.show_cursor()?;
38    Ok(())
39}

41/// Installs a panic hook that restores the terminal before printing the panic.
42fn install_panic_hook() {
43    let original_hook = panic::take_hook();
44    panic::set_hook(Box::new(move |panic_info| {
45        let _ = disable_raw_mode();
46        let _ = execute!(io::stdout(), LeaveAlternateScreen);
47        original_hook(panic_info);
48    }));
49}

51#[tokio::main]
52async fn main() -> anyhow::Result<()> {
53    // Initialize tracing (logs to file to avoid terminal interference)
54    tracing_subscriber::fmt()
55        .with_writer(|| {
56            std::fs::OpenOptions::new()
57                .create(true)
58                .append(true)
59                .open("/tmp/trefm.log")
60                .expect("failed to open log file")
61        })
62        .with_max_level(tracing::Level::DEBUG)
63        .init();
64
65    install_panic_hook();
66
67    let mut terminal = setup_terminal()?;
68
69    let start_dir = std::env::args()
70        .nth(1)
71        .map(std::path::PathBuf::from)
72        .unwrap_or_else(|| std::env::current_dir().expect("failed to get current directory"));
73
74    let result = run_app(&mut terminal, &start_dir).await;
75
76    restore_terminal(&mut terminal)?;
77
78    if let Err(e) = result {
79        eprintln!("Error: {e}");
80        std::process::exit(1);
81    }
82
83    Ok(())
84}

86async fn run_app(
87    terminal: &mut Terminal<CrosstermBackend<io::Stdout>>,
88    start_dir: &std::path::Path,
89) -> anyhow::Result<()> {
90    let mut app = App::new(start_dir)?;
91    let mut input_state = InputState::new();
92
93    loop {
94        terminal.draw(|f| render(f, &app))?;
95
96        if app.should_quit() {
97            break;
98        }
99
100        // Poll for crossterm events
101        if event::poll(std::time::Duration::from_millis(100))? {
102            if let Event::Key(key) = event::read()? {
103                let (action, new_input_state) = handle_key(key, app.mode(), &input_state);
104                input_state = new_input_state;
105
106                app = match action {
107                    InputAction::Command(cmd) => app.with_clear_status().handle_command(cmd),
108                    InputAction::EnterMode(mode) => app.with_mode(mode),
109                    InputAction::Quit => app.with_quit(),
110                    InputAction::CursorTop => {
111                        let new_panel = app.panel().clone().with_cursor_top();
112                        app.with_panel(new_panel)
113                    }
114                    InputAction::CursorBottom => {
115                        let new_panel = app.panel().clone().with_cursor_bottom();
116                        app.with_panel(new_panel)
117                    }
118                    InputAction::NextSort => match app.panel().with_next_sort() {
119                        Ok(new_panel) => {
120                            let msg = format!("Sort: {:?}", new_panel.sort_field());
121                            app.with_panel(new_panel).with_status(msg)
122                        }
123                        Err(e) => app.with_status(format!("Error: {e}")),
124                    },
125                    InputAction::RequestDelete => {
126                        if let Some(entry) = app.panel().selected_entry() {
127                            let path = entry.path().to_path_buf();
128                            app.with_mode(AppMode::Confirm(ConfirmAction::Delete(vec![path])))
129                        } else {
130                            app
131                        }
132                    }
133                    InputAction::ConfirmApproved => handle_confirm_approved(app),
134                    InputAction::None => app,
135                };
136            }
137        }
138    }
139
140    Ok(())
141}

143fn handle_confirm_approved(app: App) -> App {
144    match app.mode().clone() {
145        AppMode::Confirm(ConfirmAction::Delete(paths)) => {
146            for path in &paths {
147                if let Err(e) = trefm_core::delete_file(path) {
148                    return app
149                        .with_mode(AppMode::Normal)
150                        .with_status(format!("Delete failed: {e}"));
151                }
152            }
153            let msg = format!("Deleted {} item(s)", paths.len());
154            match app.panel().refresh() {
155                Ok(new_panel) => app
156                    .with_mode(AppMode::Normal)
157                    .with_panel(new_panel)
158                    .with_status(msg),
159                Err(e) => app
160                    .with_mode(AppMode::Normal)
161                    .with_status(format!("Error refreshing: {e}")),
162            }
163        }
164        _ => app.with_mode(AppMode::Normal),
165    }
166}
