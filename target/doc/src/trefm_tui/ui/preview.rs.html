1//! Preview panel rendering with syntax highlighting.
2//!
3//! Displays a context-aware preview for the currently selected entry:
4//! text files are syntax-highlighted via `syntect`, directories show
5//! an indented tree snapshot, and binary files show a size summary.
6
7use std::path::Path;
8use std::sync::OnceLock;
9
10use ratatui::{
11    layout::Rect,
12    style::{Color, Modifier, Style},
13    text::{Line, Span},
14    widgets::{Block, Borders, Paragraph},
15    Frame,
16};
17use syntect::highlighting::{Theme, ThemeSet};
18use syntect::parsing::SyntaxSet;
19use trefm_core::fs::entry::FileEntry;
20use trefm_core::fs::preview::{is_binary, read_directory_tree, read_text_preview};

22/// Lazily initialised syntax set.
23fn syntax_set() -> &'static SyntaxSet {
24    static SS: OnceLock<SyntaxSet> = OnceLock::new();
25    SS.get_or_init(SyntaxSet::load_defaults_newlines)
26}
27
28/// Lazily initialised highlighting theme.
29fn theme() -> &'static Theme {
30    static TH: OnceLock<Theme> = OnceLock::new();
31    TH.get_or_init(|| {
32        let ts = ThemeSet::load_defaults();
33        ts.themes
34            .get("base16-ocean.dark")
35            .cloned()
36            .unwrap_or_else(|| {
37                ts.themes
38                    .into_values()
39                    .next()
40                    .expect("syntect ships at least one theme")
41            })
42    })
43}

45/// Renders the preview panel for the currently selected entry.
46///
47/// - **Directories**: shows an indented tree snapshot (depth 3, max 50 entries).
48/// - **Text files**: shows syntax-highlighted content with line numbers.
49/// - **Binary files**: shows a size message.
50/// - **No selection**: shows a placeholder.
51pub fn render_preview(f: &mut Frame, area: Rect, selected: Option<&FileEntry>) {
52    let content = match selected {
53        Some(entry) if entry.is_dir() => render_directory_preview(entry.path()),
54        Some(entry) => render_file_preview(entry),
55        None => vec![Line::from(Span::styled(
56            "No file selected",
57            Style::default().fg(Color::DarkGray),
58        ))],
59    };
60
61    let block = Block::default()
62        .borders(Borders::ALL)
63        .title("Preview")
64        .border_style(Style::default().fg(Color::DarkGray));
65
66    let paragraph = Paragraph::new(content).block(block);
67    f.render_widget(paragraph, area);
68}

70/// Renders a directory tree preview.
71fn render_directory_preview(path: &Path) -> Vec<Line<'static>> {
72    let mut lines: Vec<Line<'static>> = vec![
73        Line::from(Span::styled(
74            format!(
75                "Directory: {}",
76                path.file_name()
77                    .map(|n| n.to_string_lossy().into_owned())
78                    .unwrap_or_else(|| path.display().to_string())
79            ),
80            Style::default()
81                .fg(Color::Blue)
82                .add_modifier(Modifier::BOLD),
83        )),
84        Line::from(""),
85    ];
86
87    match read_directory_tree(path, 3, 50) {
88        Ok(entries) if entries.is_empty() => {
89            lines.push(Line::from(Span::styled(
90                "  (empty directory)",
91                Style::default().fg(Color::DarkGray),
92            )));
93        }
94        Ok(entries) => {
95            for entry in &entries {
96                let indent = "  ".repeat(entry.depth + 1);
97                let (icon, color) = if entry.is_dir {
98                    ("/", Color::Blue)
99                } else {
100                    (" ", Color::White)
101                };
102                lines.push(Line::from(Span::styled(
103                    format!("{indent}{icon}{}", entry.name),
104                    Style::default().fg(color),
105                )));
106            }
107        }
108        Err(_) => {
109            lines.push(Line::from(Span::styled(
110                "  (unable to read directory)",
111                Style::default().fg(Color::Red),
112            )));
113        }
114    }
115
116    lines
117}

119/// Renders a file preview — syntax-highlighted text or binary message.
120fn render_file_preview(entry: &FileEntry) -> Vec<Line<'static>> {
121    let path = entry.path();
122
123    // Check if binary
124    match is_binary(path) {
125        Ok(true) => {
126            return vec![Line::from(Span::styled(
127                format!("Binary file - {}", format_preview_size(entry.size())),
128                Style::default().fg(Color::DarkGray),
129            ))];
130        }
131        Err(_) => {
132            return vec![Line::from(Span::styled(
133                "Unable to read file",
134                Style::default().fg(Color::Red),
135            ))];
136        }
137        Ok(false) => {}
138    }
139
140    // Read text preview
141    let preview = match read_text_preview(path, 100) {
142        Ok(p) => p,
143        Err(_) => {
144            return vec![Line::from(Span::styled(
145                "Unable to read file",
146                Style::default().fg(Color::Red),
147            ))];
148        }
149    };
150
151    // Apply syntax highlighting
152    let highlighted = highlight_lines(path, &preview.lines);
153
154    let line_number_width = preview.total_lines.to_string().len().max(3);
155
156    let mut output: Vec<Line<'static>> = highlighted
157        .into_iter()
158        .enumerate()
159        .map(|(i, spans)| {
160            let line_num = format!("{:>width$} ", i + 1, width = line_number_width);
161            let mut all_spans = vec![Span::styled(line_num, Style::default().fg(Color::DarkGray))];
162            all_spans.extend(spans);
163            Line::from(all_spans)
164        })
165        .collect();
166
167    if preview.is_truncated {
168        output.push(Line::from(""));
169        output.push(Line::from(Span::styled(
170            format!(
171                "[truncated — showing {}/{} lines]",
172                preview.lines.len(),
173                preview.total_lines
174            ),
175            Style::default()
176                .fg(Color::Yellow)
177                .add_modifier(Modifier::ITALIC),
178        )));
179    }
180
181    output
182}

184/// Applies syntect highlighting to lines, converting to ratatui Spans.
185fn highlight_lines(path: &Path, lines: &[String]) -> Vec<Vec<Span<'static>>> {
186    let ss = syntax_set();
187    let th = theme();
188
189    let syntax = path
190        .extension()
191        .and_then(|ext| ss.find_syntax_by_extension(&ext.to_string_lossy()))
192        .or_else(|| {
193            path.file_name()
194                .and_then(|name| ss.find_syntax_by_extension(&name.to_string_lossy()))
195        })
196        .unwrap_or_else(|| ss.find_syntax_plain_text());
197
198    let mut highlighter = syntect::easy::HighlightLines::new(syntax, th);
199
200    lines
201        .iter()
202        .map(|line| {
203            let regions = highlighter.highlight_line(line, ss).unwrap_or_default();
204
205            regions
206                .into_iter()
207                .map(|(style, text)| {
208                    let fg = Color::Rgb(style.foreground.r, style.foreground.g, style.foreground.b);
209                    Span::styled(text.to_string(), Style::default().fg(fg))
210                })
211                .collect()
212        })
213        .collect()
214}

216fn format_preview_size(bytes: u64) -> String {
217    const KB: u64 = 1024;
218    const MB: u64 = KB * 1024;
219    const GB: u64 = MB * 1024;
220
221    if bytes >= GB {
222        format!("{:.1} GB", bytes as f64 / GB as f64)
223    } else if bytes >= MB {
224        format!("{:.1} MB", bytes as f64 / MB as f64)
225    } else if bytes >= KB {
226        format!("{:.1} KB", bytes as f64 / KB as f64)
227    } else {
228        format!("{bytes} B")
229    }
