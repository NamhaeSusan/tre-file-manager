1//! Key binding configuration.
2//!
3//! Key bindings map key names (e.g. `"j"`, `"gg"`, `"Enter"`) to action
4//! identifiers (e.g. `"cursor_down"`, `"go_first"`, `"open"`).
5//! The default bindings follow vim conventions.
6
7use std::collections::HashMap;
8use std::path::Path;
9
10use serde::{Deserialize, Serialize};
11
12use crate::error::{CoreError, CoreResult};
13
14/// A single key-to-action mapping (used for serialization).
15#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
16pub struct KeyBinding {
17    /// The key name (e.g. `"j"`, `"Enter"`, `"Space"`).
18    pub key: String,
19    /// The action identifier (e.g. `"cursor_down"`, `"open"`).
20    pub action: String,
21}
22
23/// Complete set of key bindings.
24///
25/// Stores bindings as a `HashMap<String, String>` for O(1) lookup.
26/// The default instance provides vim-style navigation.
27#[derive(Debug, Clone, Serialize, Deserialize)]
28pub struct Keymap {
29    #[serde(default)]
30    pub bindings: HashMap<String, String>,
31}
32
33impl Default for Keymap {
34    fn default() -> Self {
35        let mut bindings = HashMap::new();
36
37        // Navigation
38        bindings.insert("j".to_string(), "cursor_down".to_string());
39        bindings.insert("k".to_string(), "cursor_up".to_string());
40        bindings.insert("h".to_string(), "go_parent".to_string());
41        bindings.insert("l".to_string(), "enter_dir".to_string());
42        bindings.insert("gg".to_string(), "go_first".to_string());
43        bindings.insert("G".to_string(), "go_last".to_string());
44        bindings.insert("Enter".to_string(), "open".to_string());
45
46        // File operations
47        bindings.insert("y".to_string(), "yank".to_string());
48        bindings.insert("d".to_string(), "delete".to_string());
49        bindings.insert("p".to_string(), "paste".to_string());
50        bindings.insert("r".to_string(), "rename".to_string());
51
52        // Toggles and search
53        bindings.insert(".".to_string(), "toggle_hidden".to_string());
54        bindings.insert("/".to_string(), "search".to_string());
55        bindings.insert("Space".to_string(), "select_toggle".to_string());
56        bindings.insert("s".to_string(), "sort_cycle".to_string());
57
58        // Bookmarks
59        bindings.insert("b".to_string(), "bookmark_add".to_string());
60        bindings.insert("'".to_string(), "bookmark_go".to_string());
61
62        // Panels and misc
63        bindings.insert("Tab".to_string(), "panel_switch".to_string());
64        bindings.insert("q".to_string(), "quit".to_string());
65        bindings.insert("?".to_string(), "help".to_string());
66
67        Self { bindings }
68    }
69}
70
71impl Keymap {
72    /// Loads key bindings from a TOML file at `path`.
73    ///
74    /// # Errors
75    ///
76    /// - [`CoreError::NotFound`] if the file does not exist.
77    /// - [`CoreError::PermissionDenied`] if the file is not readable.
78    /// - [`CoreError::ConfigParse`] if the TOML is malformed.
79    pub fn load(path: &Path) -> CoreResult<Self> {
80        let content = std::fs::read_to_string(path).map_err(|e| match e.kind() {
81            std::io::ErrorKind::NotFound => CoreError::NotFound(path.to_path_buf()),
82            std::io::ErrorKind::PermissionDenied => CoreError::PermissionDenied(path.to_path_buf()),
83            _ => CoreError::Io(e),
84        })?;
85        toml::from_str(&content).map_err(|e| CoreError::ConfigParse(e.to_string()))
86    }
87
88    /// Returns the action mapped to `key`, or `None` if unbound.
89    pub fn action_for_key(&self, key: &str) -> Option<&str> {
90        self.bindings.get(key).map(|s| s.as_str())
91    }
92}

152    #[test]
153    fn load_custom_keymap() {
154        let tmp = TempDir::new().unwrap();
155        let path = tmp.path().join("keymap.toml");
156        fs::write(
157            &path,
158            r#"
159[bindings]
160j = "custom_down"
161k = "custom_up"
162x = "custom_action"
163"#,
164        )
165        .unwrap();
166
167        let keymap = Keymap::load(&path).unwrap();
168
169        assert_eq!(keymap.action_for_key("j"), Some("custom_down"));
170        assert_eq!(keymap.action_for_key("k"), Some("custom_up"));
171        assert_eq!(keymap.action_for_key("x"), Some("custom_action"));
172        // Default keys not present since this is a fresh load (no merging with defaults)
173        assert_eq!(keymap.action_for_key("h"), None);
174    }
