8macro_rules! impl_try_into_bytes {
9    ($struct:ty) => {
10        impl TryInto<Vec<u8>> for $struct {
11            type Error = Error;
12
13            fn try_into(self) -> Result<Vec<u8>, Self::Error> {
14                ser::to_bytes(&self).map(|b| b.to_vec())
15            }
16        }
17    };
18}
